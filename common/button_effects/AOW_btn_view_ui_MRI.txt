#打开
AOW_btn_view_ui_mri_open_console = {
	potential = {
		is_scope_type = megastructure
		exists = owner
		owner = { is_same_value = from }
		NOT = { has_megastructure_flag = AOW_flag_MEGA_open_mri_console }
		OR = {
			is_megastructure_type = AOW_MEGA_material_recombination_instrument_1
			is_megastructure_type = AOW_MEGA_material_recombination_instrument_2
			is_megastructure_type = AOW_MEGA_material_recombination_instrument_3
			is_megastructure_type = AOW_MEGA_material_recombination_instrument_4
			is_megastructure_type = AOW_MEGA_material_recombination_instrument_5
		}
	}
	allow = { always = yes }
	effect = {
		custom_tooltip = AOW_text_mri_open_console
		set_megastructure_flag = AOW_flag_MEGA_open_mri_console
	}
}

#复制产出
AOW_btn_view_ui_mri_copy = {
	potential = {
		is_scope_type = megastructure
		exists = owner
		owner = { is_same_value = from }
		has_megastructure_flag = AOW_flag_MEGA_open_mri_console
	}
	allow = { always = yes }
	effect = {
		custom_tooltip = AOW_text_mri_copy_desc
		hidden_effect = {
			save_event_target_as = AOW_targert_mega
			owner = {
				random_owned_megastructure = {
					limit = {
						NOT = { is_same_value = event_target:AOW_targert_mega }
						OR = {
							is_megastructure_type = AOW_MEGA_material_recombination_instrument_1
							is_megastructure_type = AOW_MEGA_material_recombination_instrument_2
							is_megastructure_type = AOW_MEGA_material_recombination_instrument_3
							is_megastructure_type = AOW_MEGA_material_recombination_instrument_4
							is_megastructure_type = AOW_MEGA_material_recombination_instrument_5
						}
						OR = {
							check_variable = { which = AOW_var_mri_points_alloys value != event_target:AOW_targert_mega.AOW_var_mri_points_alloys }
							check_variable = { which = AOW_var_mri_points_cg value != event_target:AOW_targert_mega.AOW_var_mri_points_cg }
							check_variable = { which = AOW_var_mri_points_vm value != event_target:AOW_targert_mega.AOW_var_mri_points_vm }
							check_variable = { which = AOW_var_mri_points_eg value != event_target:AOW_targert_mega.AOW_var_mri_points_eg }
							check_variable = { which = AOW_var_mri_points_rc value != event_target:AOW_targert_mega.AOW_var_mri_points_rc }
							check_variable = { which = AOW_var_mri_points_lm value != event_target:AOW_targert_mega.AOW_var_mri_points_lm }
							check_variable = { which = AOW_var_mri_points_zro value != event_target:AOW_targert_mega.AOW_var_mri_points_zro }
							check_variable = { which = AOW_var_mri_points_dm value != event_target:AOW_targert_mega.AOW_var_mri_points_dm }
							check_variable = { which = AOW_var_mri_points_nanites value != event_target:AOW_targert_mega.AOW_var_mri_points_nanites }
						}
					}
					set_variable = { which = AOW_var_mri_points_alloys value = event_target:AOW_targert_mega.AOW_var_mri_points_alloys }
					set_variable = { which = AOW_var_mri_points_cg value = event_target:AOW_targert_mega.AOW_var_mri_points_cg }
					set_variable = { which = AOW_var_mri_points_vm value = event_target:AOW_targert_mega.AOW_var_mri_points_vm }
					set_variable = { which = AOW_var_mri_points_eg value = event_target:AOW_targert_mega.AOW_var_mri_points_eg }
					set_variable = { which = AOW_var_mri_points_rc value = event_target:AOW_targert_mega.AOW_var_mri_points_rc }
					set_variable = { which = AOW_var_mri_points_lm value = event_target:AOW_targert_mega.AOW_var_mri_points_lm }
					set_variable = { which = AOW_var_mri_points_zro value = event_target:AOW_targert_mega.AOW_var_mri_points_zro }
					set_variable = { which = AOW_var_mri_points_dm value = event_target:AOW_targert_mega.AOW_var_mri_points_dm }
					set_variable = { which = AOW_var_mri_points_nanites value = event_target:AOW_targert_mega.AOW_var_mri_points_nanites }
				}
			}
		}
	}
}

		custom_tooltip = {#on_build_start

			from = { NOT = { has_country_flag = AOW_flag_CNTR_no_more_deep_space_citadel } }
		}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {

		modifier = {

	on_build_start = {
		from = {
			if = {
				limit = {
					is_ai = no
					event_target:global_event_country = {
						check_variable = { which = AOW_var_player_mega_cap_deep_space_citadel value > 0 }
					}
				}
				if = {
					limit = {
						root = {
							count_ship_in_system = {
								limit = {
									is_space_citadel_ship_size = yes

						}
					}
					set_country_flag = AOW_flag_CNTR_no_more_deep_space_citadel
				}
			}
		}
	}

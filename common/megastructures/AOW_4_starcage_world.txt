				trigger_megastructure_icon = yes

				if = {
					limit = {
						root.star = {
							OR = {
								is_planet_class = pc_g_star
								is_planet_class = pc_k_star
								is_planet_class = pc_m_star
								is_planet_class = pc_d_star
							}
						}
					}
					change_pc = pc_starcage_world_common_star
					set_name = "pc_starcage_world_common_star"
					add_district = AOW_district_sw_common_star
				}
				if = {
					limit = {
						root.star = {
							AOW_trgr_PLNT_composed_of_neutron = yes
						}
					}
					change_pc = pc_starcage_world_neutron_star
					set_name = "pc_starcage_world_neutron_star"
					add_district = AOW_district_sw_neutron_star
				}
				if = {
					limit = {
						root.star = {
							is_planet_class = pc_black_hole
						}
					}
					change_pc = pc_starcage_world_black_hole
					set_name = "pc_starcage_world_black_hole"
					add_district = AOW_district_sw_black_hole
				}
				set_planet_entity = { entity = AOW_empty_entity }
				add_deposit = AOW_d_starcage_world
				set_owner = fromfrom.owner
				set_controller = fromfrom.owner
				create_pop_group = {
					species = owner_main_species
				}
				validate_planet_buildings_and_districts = yes
			}
		}
	}

		}
		random_system_planet = {
			limit = { has_planet_flag = AOW_flag_PLNT_origin_starcage_world }
			set_planet_flag = AOW_flag_PLNT_starcage_world_@owner
			if = {
				limit = { NOT = { has_deposit = AOW_d_starcage_world } }
				add_deposit = AOW_d_starcage_world

			if = {#恒星
				limit = {
					solar_system = { has_star_flag = AOW_flag_SSTM_origin_starcage_world_common_star }
				}
				change_pc = pc_starcage_world_common_star
				export_trigger_value_to_variable = {
					trigger = num_districts
					parameters = { type = AOW_district_sw_common_star }
					variable = AOW_var_district_sw_common_star
				}
				while = {
					count = value:AOW_var_starcage_world_old_districts|v|AOW_var_district_sw_common_star|
					remove_district = AOW_district_sw_common_star
				}
				while = {
					count = value:AOW_var_starcage_world_new_districts|v|AOW_var_district_sw_common_star|
					add_district = AOW_district_sw_common_star
				}
				clear_variable = AOW_var_district_sw_common_star
			}
			if = {#中子
				limit = {
					solar_system = { has_star_flag = AOW_flag_SSTM_origin_starcage_world_neutron_star }
				}
				change_pc = pc_starcage_world_neutron_star
				remove_modifier = AOW_mod_PLNT_neutron_star_magnetic_field
				export_trigger_value_to_variable = {
					trigger = num_districts
					parameters = { type = AOW_district_sw_neutron_star }
					variable = AOW_var_district_sw_neutron_star
				}
				while = {
					count = value:AOW_var_starcage_world_old_districts|v|AOW_var_district_sw_neutron_star|
					remove_district = AOW_district_sw_neutron_star
				}
				while = {
					count = value:AOW_var_starcage_world_new_districts|v|AOW_var_district_sw_neutron_star|
					add_district = AOW_district_sw_neutron_star
				}
				clear_variable = AOW_var_district_sw_neutron_star
			}
			
			export_trigger_value_to_variable = {
				trigger = num_districts
				parameters = { type = AOW_district_starcage_world_1 }
				variable = AOW_var_district_starcage_world_1
			}
			while = {
				count = value:AOW_var_starcage_world_old_districts|v|AOW_var_district_starcage_world_1|
				remove_district = AOW_district_starcage_world_1
			}
			while = {
				count = value:AOW_var_starcage_world_new_districts|v|AOW_var_district_starcage_world_1|
				add_district = AOW_district_starcage_world_1
			}
			clear_variable = AOW_var_district_starcage_world_1
			
			export_trigger_value_to_variable = {
				trigger = num_districts
				parameters = { type = AOW_district_starcage_world_2 }
				variable = AOW_var_district_starcage_world_2

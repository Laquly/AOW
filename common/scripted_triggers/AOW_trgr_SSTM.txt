#星系有宜居星球
AOW_trgr_SSTM_has_habitable_planets = {
	any_system_planet = {
		is_star = no
		OR = {
			is_colonizable = yes
			is_colony = yes
		}
	}
}
#星系有非人工、非土著的殖民地
AOW_trgr_SSTM_has_non_artificial_non_primitive_colony = {
	any_system_planet = {
		OR = {
			has_colony_progress > 0
			is_colony = yes
		}
		is_artificial = no
		exists = owner
		owner = { is_primitive = no }
	}
}
#可以侵略性干扰土著
AOW_trgr_SSTM_can_aggressively_interfere_with_pre_ftl = {
	if = {
		limit = {
			any_system_planet = {
				exists = owner
				owner = { is_primitive = yes }
			}
		}
		from = { has_policy_flag = interference_aggressive }
	}
}

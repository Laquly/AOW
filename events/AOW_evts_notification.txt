			if = {#如果没有超空间航道

				set_spawn_system_batch = begin
				add_hyperlane = {
					from = event_target:AOW_tgt_SSTM_hc_start
					to = event_target:AOW_tgt_SSTM_hc_end
				}
				set_spawn_system_batch = end
				root = { set_country_flag = AOW_flag_CNTR_add_hyperlane_with_HC }
			}
			else_if = {#如果有超空间航道
				limit = { event_target:AOW_tgt_SSTM_hc_end = { has_hyperlane_to = event_target:AOW_tgt_SSTM_hc_start } }
				event_target:AOW_tgt_SSTM_hc_end = { AOW_eft_SSTM_destroy_hyper_relay = yes }
				event_target:AOW_tgt_SSTM_hc_start = { AOW_eft_SSTM_destroy_hyper_relay = yes }
				set_spawn_system_batch = begin
				remove_hyperlane = {
					from = event_target:AOW_tgt_SSTM_hc_start
					to = event_target:AOW_tgt_SSTM_hc_end
				}
				set_spawn_system_batch = end
				root = { set_country_flag = AOW_flag_CNTR_remove_hyperlane_with_HC }
			}
		}
		fromfrom = { country_event = { id = AOW_evt_hyperlane_controller.102 } }
		remove_megastructure = event_target:AOW_tgt_MEGA_hc_start
		remove_megastructure = event_target:AOW_tgt_MEGA_hc_end
		fromfrom = { add_resource = { alloys = 5000 } }
	}
}
